140/*
141 * Common piece of code to perform a sleeping.
142 *
143 * @param zzz the amount of time to
144 * sleep, expressed as nanoseconds.
145 *
146 * Example
147 SLEEP(5000000L)
148 *
149 */
150#define SLEEP(zzz) \
151 do \
152 { \
153 struct timespec ts_private; \
154 ts_private.tv_sec = 0; \
155 ts_private.tv_nsec = zzz; \
156 nanosleep(&ts_private, NULL); \
157 } while (0);
158
159/*
160 * Commonly used block of code to sleep
161 * for a specified amount of time and
162 * then jump back to a specified label.
163 *
164 * @param zzz how much time to sleep (as long nanoseconds)
165 * @param goto_to the label to which jump to
166 *
167 * Example:
168 *
169 ...
170 else
171 SLEEP_AND_GOTO(100000L, retry)
172 */
173#define SLEEP_AND_GOTO(zzz, goto_to) \
174 do \
175 { \
176 struct timespec ts_private; \
177 ts_private.tv_sec = 0; \
178 ts_private.tv_nsec = zzz; \
179 nanosleep(&ts_private, NULL); \
180 goto goto_to; \
181 } while (0);
