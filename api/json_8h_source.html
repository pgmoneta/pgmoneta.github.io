93struct json_value
94{
95 enum json_value_type type;
96 uint32_t length;
97 // Payload can hold a 64bit value such as an integer, or be a pointer to a string, float or an array
98 // Based on value type, payload could be converted to the following:
99 // int64 (ValueInt), int64*(ValueInt64Array), char*(ValueString),
100 // char**(ValueStringArray), float*(ValueFloat/ValueFloatArray),
101 // json*(ValueObject), json**(ValueItemArray/ValueArrayArray)
102 void* payload;
103};

108struct json
109{
110 // a json object can only be item or array
111 enum json_type type;
112 // if the object is an array, it can have at most one json element
113 struct json_element* element;
114};

119struct json_element
120{
121 // if the element holds an array value,
122 // if it belongs to a json array object, then the key is NULL, and it will not have next element,
123 // if it belongs to a json item object, then the key is not NULL, and it probably has next element
124 char* key;
125 struct json_value* value;
126 struct json_element* next;
127};
